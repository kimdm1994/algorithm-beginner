    public static String resolve(String token) {
        Stack<Character> stack = new Stack<>();
        boolean isVPS = true;

        for (int i = 0; i < token.length(); i++) {
            char c = token.charAt(i);

            if (c == '(') {
                stack.push(c);
            } else if (stack.isEmpty()) {
                return "NO";
            } else {
                stack.pop();
            }
        }

        if (stack.isEmpty()) {
            isVPS = true;
            return "YES";
        } else {
            isVPS = false;
            return "NO";
        }
    }
}
